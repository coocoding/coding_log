#include<iostream>
using namespace std;

int main() {
	int n;
	cin >> n;
	int arr[1001];
	int dp[1001];
	int result = 0;
	for (int i = 1; i <= n; i++) {
		cin >> arr[i];
	}
	for (int i = 1; i <= n; i++) {
		dp[i] = arr[i];
		for (int j = 1; j < i; j++) {
			if (arr[j] < arr[i] && dp[i] < dp[j] + arr[i])
				dp[i] = dp[j] + arr[i];
		}
		result = max(result, dp[i]);
	}
	cout << result;
}
